# Steps
As explained in above YouTube video

git clone https://github.com/imademethink/LighterRAG_projects.git

cd LighterRAG_projects-main

# For Text processing

cd P1_Text

pip install -r requirements.txt

python light_rag_text_0.py

# For Image processing

cd P2_Image

pip install -r requirements.txt

python light_rag_img_0.py

# For Audio processing

cd P3_Audio

pip install -r requirements.txt

python light_rag_aud_0.py
